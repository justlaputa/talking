Let's talk about Scope


var foo = "foo";

function scope() {
    var bar = "bar";

    console.log("I can see foo:", foo);       //OK
    console.log("I can also see bar:", bar);  //Also OK
}

scope();

console.log("foo is a global variable:", foo); //OK, foo is global
console.log("but bar is local:", bar); //ReferenceError: bar is not defined

Use var


function scope() {
    foo = 1;
}

scope();

console.log(foo); // 1
                    

function scope() {
    var foo = 1;
    console.log(foo);
}

scope();

console.log(foo); // ReferenceError: foo is not defined

Block scope VS. Function scope

In Java(C, C++)


public class Scope {
    public static void main(String[] args) {
        if(true) {
            String inBlock = "i'm in block";
            System.out.println(inBlock);
        }

        System.out.println(inBlock); //cannot find symbol
    }
}

In Javascript


function scope() {
    if(1) {
        var in_block = "I'm in block";
        console.log(in_block);  // I'm in block
    }

    console.log(in_block);  // I'm in block
}

scope();

Function Scope?


function scope() {

  function inner() {
      var inFunction = "I'm in function";
      
      console.log(inFunction);  //I'm in function
  }
  
  inner();
  
  console.log(inFunction); //ReferenceError
}

scope();

Hoisting?


var foo = 1;
function hoist() {
    foo = 100;
    console.log(foo);
    return;
    var foo;
}
hoist();
console.log(foo); // 100 ? 1 ?
                    

var foo = 1;
function hoist() {
    var foo;
    foo = 100;
    console.log(foo);
    return;
}
hoist();
console.log(foo); // 1

What's happened?


function foo() {
    bar();
    var x = 1;
}
                    

function foo() {
    var x;

    x = 1;
}

Even if it never executes


function foo() {
    if (false) {
        var x = 1;
    }
    return;
    var y = 1;
}
                    

function foo() {
    var x, y;
    if (false) {
        x = 1;
    }
    return;
    y = 1;
}

How about functions?


function hoist() {
    foo();
    bar();
    var foo = function() {
        console.log("in foo");
    };
    function bar() {
        console.log("in bar");
    }
}

hoist();
                    

function hoist() {
    var foo;
    function bar() {
        console.log("in bar");
    }
    foo();
    bar();
    foo = function() {
        console.log("in foo");
    };
}

hoist();

Now, what should we do?

  • Declare all variable with one var
  • And declare at the begining of the scope
