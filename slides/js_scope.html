Let's talk about Scope


var foo = "foo";

function scope() {
    var bar = "bar";

    console.log("I can see foo:", foo);
    console.log("I can also see bar:", bar);
}

scope();

console.log("foo is a global variable:", foo);
console.log("but bar is local:", bar);

I can see foo: foo

I can also see bar: bar

foo is a global variable: foo

ReferenceError: bar is not defined

Use var


function scope() {
    foo = 1;
}

scope();

console.log(foo); // 1
					

function scope() {
    var foo = 1;
    console.log(foo);
}

scope();

console.log(foo); // ReferenceError: foo is not defined
